# useMediaQuery
dep: https://github.com/streamich/use-media

## Usage

```jsx
import React from 'react';
import { useMediaQuery } from '@qunhe/react-hooks-collection';

const Com = () => {
  const matches = useMediaQuery('(min-width: 576px)');

  return (
    <div>
      matches: {matches}
    </div>
  );
};
```
