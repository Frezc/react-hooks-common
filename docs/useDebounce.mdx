# usePromise

## API
same as https://lodash.com/docs/4.17.11#debounce

## Usage

```js
import React from 'react';
import { useDebounce } from '@qunhe/rhc';

const Com = () => {
  const debouncedChange = useDebounce((v) => {
    postUpdate();
  }, 500, { maxWait: 2000 });

  return (
    <input
      type="text"
      onChange={(e) => {
        debouncedChange(e.target.value)
      }}
    />
  )
}
```
