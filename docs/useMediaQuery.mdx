# useMediaQuery
A media query hook.
dep: https://github.com/streamich/use-media

## API
```typescript
function useMedia(
  rawQuery: string | MediaQueryObject,
  defaultState: boolean = false
) : boolean;
```
- `rawQuery`: Regular query string `"(min-width: 400px)"` or object like `{ minWidth: 400 }` is available.
- `defaultState`: Value before component mounted.
- **return**: matches

## Usage

```jsx
import React from 'react';
import { useMediaQuery } from 'react-hooks-common';

const Com = () => {
  const matches = useMediaQuery('(min-width: 576px)');

  return (
    <div>
      matches: {matches}
    </div>
  );
};
```
