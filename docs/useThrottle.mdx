# useThrottle
A wrapper of throttle of lodash.

## API
same as https://lodash.com/docs/4.17.11#throttle

## Usage

```js
import React from 'react';
import { useThrottle } from 'react-hooks-common';

const Com = () => {
  const debouncedChange = useThrottle((v) => {
    sendMessage();
  }, 500, { leading: false });

  return (
    <input
      type="text"
      onChange={(e) => {
        debouncedChange(e.target.value)
      }}
    />
  )
}
```
